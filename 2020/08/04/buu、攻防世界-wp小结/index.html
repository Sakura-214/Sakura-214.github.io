EJ>CvSHMV7G9R9@?3k

发现错误，观察得到，前四位依次加上对应位数字，即e+1=f ， j+2=l ， >+3=a , c+4=g ;

那么编写脚本如下：

#include<iostream>
#include<cstring>

using namespace std;

int main()
{
    int a[18] = { 69 ,74 ,62, 67 ,118, 83, 72, 77 ,86 ,55 ,71 ,57, 82 ,57 ,64 ,63, 51 ,107 };
    for (int i = 1; i <= 18; i++)
    {
        a[i - 1] += i;
        if (a[i - 1] >= 'A' && a[i - 1] <= 'Z')
        {
            a[i - 1] += 32;
        }
        cout << (char)a[i - 1];
    }

    return 0;
}

    temp1 = k.replace('?',str(chr(65+i)),1)

        temp2 = temp1.replace('?',chr(65+j),1)
        for n in range(26):
            temp3 = temp2.replace('?',chr(65+n),1)
