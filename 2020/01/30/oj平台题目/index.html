{

{
    scanf("%d",&n[i]);

{
    p = 0;
    x = n[i];
    if (x == 0 || x == 1||x==3)
    {
        printf("No\n");
        continue;
    }
    if (x == 2)
    {
        printf("Yes\n");
        continue;
    }
    for (int i1 = 2; i1 <= x - 1; i1++)
    {
        if (x % i1 == 0)
        {
            a = i1;
            b = x / i1;
            h = 0;
            k = 0;
            for (int i2 = 2; i2 <= a - 1; i2++)
            {
                if (a % i2 != 0)
                {
                    h++;
                }
            }
            for (int i2 = 2; i2 <= b - 1; i2++)
            {
                if (b % i2 != 0)
                {
                    k++;
                }
            }

                break;
            }
            else
            {
                printf("No\n");
                break;
            }
        }

        {
            p++;
        }

}

}
